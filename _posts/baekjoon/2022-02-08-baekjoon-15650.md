### 문제 풀이

```python
n, m = map(int, input().strip().split())
o = []

def back():
    if len(o)==m:
        print(' '.join(map(str, o)))
        return
    for i in range(1, n+1):
        a = True
        if i not in o:
            for j in o:
                if j>i:
                    a = False
            if a:
                o.append(i)
                back()
                o.pop()

back()
```
