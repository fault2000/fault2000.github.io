### 문제 풀이

```python
n, m = map(int, input().strip().split())
o = []

def back():
    if len(o)==m:
        print(' '.join(map(str, o)))
        return
    for i in range(1, n+1):
        if i not in o:
            o.append(i)
            back()
            o.pop()

back()
```
